<prog> -> require "ifj21" <start>
<start> -> <declare-list><function-list><statement-list><start>
<start> -> eps
<declare-list> -> <declare><declare-list>
<declare-list> -> eps 
<declare> -> local <id>:<type>
<declare> -> global <id>:<type>

<function-list> -> <function><end><function-list>
<function-list> -> eps
<function> -> function <id> (<func-assign_list>) : <type> 
<func-assign-list> -> <func-assign>,<func-assign-list>
<func-assign-list> -> eps
<func-assign> -> <id>:<type>

<if-list> -> <if-statement><else-statement><end><if-list>
<if-list> -> eps
<if-statement>-> if(<expression>) then <statement-list>
<else-statement>-> else <statement-list>
<else-statement> -> eps

<while> -> <while_statement><end>
<while-statement> -> while (<statement>) do <statement-list>

<statement-list> -> <statement><statement-list>
<statement-list> -> eps
<statement> -> <assign><declare-list><expression><while><if-list><function-list><statement-list>

<assign> -> <id>=<id>
<expression> -> <id><operator><id>

<end> -> end
<type> -> integer
<type> -> string
<type> -> number
<operator> -> <
<operator> -> >
<operator> -> ==
<operator> -> ~=
<operator> -> <=
<operator> -> >=
<operator> -> +
<operator> -> -
<operator> -> /
<operator> -> //
<operator> -> *
<operator> -> ..
<operator> -> #
<operator> -> %
<operator> -> ^
